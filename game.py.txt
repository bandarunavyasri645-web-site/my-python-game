import random

print("🎮 Welcome to Guess the Number!")
number = random.randint(1, 10)
tries = 0

while True:
    guess = int(input("Guess a number between 1 and 10: "))
    tries += 1
    
    if guess < number:
        print("📉 Too low! Try again.")
    elif guess > number:
        print("📈 Too high! Try again.")
    else:
        print(f"🎉 Correct! You got it in {tries} tries!")
        break
